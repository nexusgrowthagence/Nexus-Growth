/**

 * SpotlightCard: Ajoute l'effet de halo lumineux qui suit la souris.

 * Utilisé sur toutes les cartes pour l'aspect premium.

 */

const SpotlightCard = ({

  children,

  className = "",

  onClick,

  onMouseEnter,

  onMouseLeave

}: {

  children: React.ReactNode,

  className?: string,

  onClick?: () => void,

  onMouseEnter?: () => void,

  onMouseLeave?: () => void

}) => {

  const divRef = useRef<HTMLDivElement>(null);

  const [position, setPosition] = useState({ x: 0, y: 0 });

  const [opacity, setOpacity] = useState(0);

  const handleMouseMove = (e: React.MouseEvent<HTMLDivElement>) => {

    if (!divRef.current) return;

    const rect = divRef.current.getBoundingClientRect();

    setPosition({ x: e.clientX - rect.left, y: e.clientY - rect.top });

  };

  return (

    <div

      ref={divRef}

      onMouseMove={handleMouseMove}

      onMouseEnter={() => { setOpacity(1); if (onMouseEnter) onMouseEnter(); }}

      onMouseLeave={() => { setOpacity(0); if (onMouseLeave) onMouseLeave(); }}

      onClick={onClick}

      className={`relative overflow-hidden rounded-[2rem] border border-white/5 bg-[#0F0F12] transition-colors duration-300 ${className}`}

    >

      <div

        className="pointer-events-none absolute -inset-px opacity-0 transition duration-300 z-10"

        style={{

          opacity,

          background: `radial-gradient(600px circle at ${position.x}px ${position.y}px, rgba(99, 102, 241, 0.15), transparent 40%)`,

        }}

      />

      <div className="relative z-20 h-full">

        {children}

      </div>

    </div>

  );

};

const BookingWidget = ({ bookedSlots }: { bookedSlots: any[] }) => {

  const [currentDate, setCurrentDate] = useState(new Date());

  const [selectedDay, setSelectedDay] = useState<number | null>(null);

  const [selectedTime, setSelectedTime] = useState<string | null>(null);

  const [view, setView] = useState<'calendar' | 'form' | 'success'>('calendar');

  const [isSubmitting, setIsSubmitting] = useState(false);

  const WEBHOOK_URL = "https://hook.eu1.make.com/8q2ubce9fuezm6j9ktudwwhfhldchp8t";

  const getDaysInMonth = (date: Date) => new Date(date.getFullYear(), date.getMonth() + 1, 0).getDate();

  const getFirstDayOfMonth = (date: Date) => {

    const day = new Date(date.getFullYear(), date.getMonth(), 1).getDay();

    return day === 0 ? 6 : day - 1;

  };

  const daysInMonth = getDaysInMonth(currentDate);

  const startOffset = getFirstDayOfMonth(currentDate);

  const monthName = new Intl.DateTimeFormat('fr-FR', { month: 'long', year: 'numeric' }).format(currentDate);

  const capitalizedMonthName = monthName.charAt(0).toUpperCase() + monthName.slice(1);

  const realToday = new Date();

  const isCurrentRealMonth =

    currentDate.getMonth() === realToday.getMonth() &&

    currentDate.getFullYear() === realToday.getFullYear();

  const handlePrevMonth = () => {

    const newDate = new Date(currentDate.getFullYear(), currentDate.getMonth() - 1, 1);

    if (newDate < new Date(realToday.getFullYear(), realToday.getMonth(), 1)) return;

    setCurrentDate(newDate);

    setSelectedDay(null);

  };

  const handleNextMonth = () => {

    setCurrentDate(new Date(currentDate.getFullYear(), currentDate.getMonth() + 1, 1));

    setSelectedDay(null);

  };

  const timeSlots = ["15:30", "16:00", "16:30", "17:00", "17:30", "18:00", "18:30"];

  const isSlotBooked = (day: number, timeIndex: number) => {

     const slot = timeSlots[timeIndex];

     const dateStr = `${day} ${capitalizedMonthName}`;

     if (bookedSlots.some(b => b.date === dateStr && b.time === slot)) return true;

     if (!isCurrentRealMonth) return false;

     const seed = (day * 47) + (timeIndex * 13);

     return seed % 8 === 0;

  };

  const isDayDisabled = (day: number) => {

     const date = new Date(currentDate.getFullYear(), currentDate.getMonth(), day);

     const d = date.getDay();

     return d === 0 || d === 6 || d === 1;

  };

  const handleSubmit = async (e: React.FormEvent) => {

      e.preventDefault();

      setIsSubmitting(true);

     

      const form = e.target as HTMLFormElement;

      const firstName = (form.elements.namedItem('firstName') as HTMLInputElement).value;

      const lastName = (form.elements.namedItem('lastName') as HTMLInputElement).value;

      const email = (form.elements.namedItem('email') as HTMLInputElement).value;

      const appointmentDate = `${selectedDay} ${capitalizedMonthName}`;

      const formData = {

          firstName,

          lastName,

          email,

          date: appointmentDate,

          time: selectedTime,

          source: "Landing Page Nexus"

      };

      try {

          // Envoi au Webhook Make

          await fetch(WEBHOOK_URL, {

              method: 'POST',

              headers: { 'Content-Type': 'application/json' },

              body: JSON.stringify(formData)

          });

          // Sauvegarde dans Firestore

          await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'appointments'), {

              ...formData,

              createdAt: serverTimestamp()

          });

          setView('success');

      } catch (error) {

          console.error("Erreur d'envoi", error);

      } finally {

          setIsSubmitting(false);

      }

  };

  if (view === 'success') {

    return (

        <div className="h-full flex flex-col items-center justify-center p-8 text-center animate-in fade-in zoom-in duration-500">

            <div className="w-20 h-20 bg-green-500/10 text-green-500 rounded-full flex items-center justify-center mb-6 border border-green-500/20 shadow-[0_0_30px_rgba(34,197,94,0.2)]">

                <CheckCircle2 className="w-10 h-10" />

            </div>

            <h3 className="text-2xl font-bold text-white mb-2">Demande envoyée !</h3>

            <p className="text-zinc-400 max-w-xs mx-auto text-sm font-light">

                Rendez-vous confirmé pour le <span className="text-white font-medium">{selectedDay} {capitalizedMonthName} à {selectedTime}</span>.

            </p>

        </div>

    );

  }

  if (view === 'form') {

      return (

          <div className="h-full flex flex-col p-6 animate-in slide-in-from-right duration-300">

              <button onClick={() => setView('calendar')} className="flex items-center gap-2 text-zinc-500 hover:text-white mb-8 text-sm transition-colors w-fit group">

                  <ArrowRight className="w-4 h-4 rotate-180 group-hover:-translate-x-1 transition-transform" /> Retour au calendrier

              </button>

             

              <div className="flex items-center gap-3 mb-8 pb-8 border-b border-white/5">

                 <div className="w-12 h-12 bg-indigo-500/20 rounded-xl flex items-center justify-center text-indigo-400 border border-indigo-500/30">

                    <Calendar className="w-6 h-6" />

                 </div>

                 <div>

                    <h3 className="text-white font-bold">Entretien découverte</h3>

                    <p className="text-zinc-500 text-sm font-light">{selectedDay} {capitalizedMonthName} • {selectedTime}</p>

                 </div>

              </div>

              <form className="space-y-4" onSubmit={handleSubmit}>

                  <div className="grid grid-cols-2 gap-4">

                     <div>

                        <label className="block text-xs text-zinc-500 mb-1.5 ml-1 font-bold uppercase tracking-widest">Prénom</label>

                        <input name="firstName" type="text" required className="w-full bg-zinc-900/50 border border-white/10 rounded-xl p-3 text-white text-sm focus:outline-none focus:border-indigo-500 transition-all" />

                     </div>

                     <div>

                        <label className="block text-xs text-zinc-500 mb-1.5 ml-1 font-bold uppercase tracking-widest">Nom</label>

                        <input name="lastName" type="text" required className="w-full bg-zinc-900/50 border border-white/10 rounded-xl p-3 text-white text-sm focus:outline-none focus:border-indigo-500 transition-all" />

                     </div>

                  </div>

                  <div>

                      <label className="block text-xs text-zinc-500 mb-1.5 ml-1 font-bold uppercase tracking-widest">Email professionnel</label>

                      <input name="email" type="email" required className="w-full bg-zinc-900/50 border border-white/10 rounded-xl p-3 text-white text-sm focus:outline-none focus:border-indigo-500 transition-all" />

                  </div>

                  <button type="submit" disabled={isSubmitting} className="w-full py-4 bg-white text-black font-bold rounded-xl mt-4 hover:bg-zinc-200 transition-all flex items-center justify-center gap-2 disabled:opacity-50">

                      {isSubmitting ? <div className="w-4 h-4 border-2 border-black/30 border-t-black rounded-full animate-spin"></div> : "Confirmer le RDV"}

                  </button>

              </form>

          </div>

      )

  }

  return (

    <div className="flex flex-col h-full animate-in fade-in duration-500">

        <div className="p-6 border-b border-white/5 flex justify-between items-center bg-zinc-900/30">

             <div>

                <h3 className="text-sm font-bold text-white uppercase tracking-wider">{capitalizedMonthName}</h3>

                <p className="text-zinc-500 text-[10px] font-bold uppercase tracking-widest mt-0.5">Europe/Paris</p>

             </div>

             <div className="flex gap-1">

                 <button onClick={handlePrevMonth} className="p-2 hover:bg-white/5 rounded-lg text-zinc-500 hover:text-white transition-colors"><ChevronRight className="w-4 h-4 rotate-180"/></button>

                 <button onClick={handleNextMonth} className="p-2 hover:bg-white/5 rounded-lg text-zinc-500 hover:text-white transition-colors"><ChevronRight className="w-4 h-4"/></button>

             </div>

        </div>

       

        <div className="flex flex-col md:flex-row h-full overflow-hidden">

            <div className="p-6 flex-1 bg-black/20 overflow-y-auto custom-scrollbar">

                <div className="grid grid-cols-7 gap-2 text-center text-[10px] font-bold text-zinc-600 uppercase mb-4">

                    <span>Lu</span><span>Ma</span><span>Me</span><span>Je</span><span>Ve</span><span>Sa</span><span>Di</span>

                </div>

                <div className="grid grid-cols-7 gap-2">

                    {Array.from({ length: startOffset }).map((_, i) => <div key={`empty-${i}`} className="aspect-square" />)}

                    {Array.from({ length: daysInMonth }).map((_, i) => {

                        const day = i + 1;

                        const disabled = isDayDisabled(day);

                        return (

                            <button key={day} disabled={disabled} onClick={() => !disabled && setSelectedDay(day)}

                                className={`aspect-square rounded-lg flex items-center justify-center text-sm font-medium transition-all ${

                                    disabled ? 'opacity-20 cursor-not-allowed text-zinc-700' :

                                    selectedDay === day ? 'bg-indigo-600 text-white shadow-lg' : 'text-zinc-400 hover:bg-white/10 hover:text-white'

                                }`}>{day}</button>

                        );

                    })}

                </div>

            </div>

           

            <div className={`p-4 w-full md:w-48 bg-zinc-900/50 border-l border-white/5 flex flex-col gap-2 ${!selectedDay ? 'opacity-50 pointer-events-none' : ''}`}>

                {!selectedDay ? <div className="h-full flex flex-center items-center justify-center text-zinc-600 text-[10px] font-bold uppercase tracking-widest text-center px-4 leading-relaxed">Sélectionnez un jour pour voir les horaires</div> : (

                    <div className="space-y-2 overflow-y-auto h-full pr-1 custom-scrollbar">

                        {timeSlots.map((slot, index) => {

                            const booked = isSlotBooked(selectedDay, index);

                            return (

                                <button key={slot} disabled={booked} onClick={() => !booked && (setSelectedTime(slot), setView('form'))}

                                    className={`w-full py-2.5 px-3 rounded-lg border text-sm font-medium transition-all text-center flex items-center justify-center gap-2 ${

                                        booked ? 'border-white/5 bg-zinc-900/50 text-zinc-700' :

                                        'border-white/5 bg-zinc-800/50 text-zinc-300 hover:border-indigo-500 hover:bg-indigo-500/10 hover:text-indigo-400 hover:scale-[1.02]'

                                    }`}>

                                    <span className={booked ? "line-through decoration-zinc-800 text-zinc-600" : ""}>{slot}</span>

                                    {booked && <span className="text-[9px] uppercase font-bold text-zinc-400 bg-white/5 border border-white/10 px-1 rounded">Complet</span>}

                                </button>

                            )

                        })}

                    </div>

                )}

            </div>

        </div>

    </div>

  );

};

              <button key={link.id} onClick={() => navigateTo(link.id)} className={`px-4 py-1.5 rounded-full text-xs font-medium transition-all ${activeNav === link.id ? 'bg-white/10 text-white shadow-inner' : 'text-zinc-400 hover:text-white hover:bg-white/5'}`}>{link.label}</button>

            ))}

          </div>

          <button onClick={() => navigateTo('contact')} className={`px-5 py-2 rounded-full text-xs font-bold transition-all ${currentPage === 'contact' ? 'bg-indigo-500 text-white shadow-[0_0_20px_rgba(99,102,241,0.4)]' : 'bg-white text-black hover:bg-zinc-200'}`}>Rendez-vous</button>

        </nav>

      </div>

      {/* MAIN CONTENT AREA */}

      <main className={`relative z-10 transition-all duration-300 ${isTransitioning ? 'opacity-0 scale-95 blur-sm' : 'opacity-100 scale-100 blur-0'}`}>

        {currentPage === 'home' && <HomePage navigateTo={navigateTo} setActiveNav={setActiveNav} />}

        {currentPage === 'results' && <ResultsPage navigateTo={navigateTo} />}

        {currentPage === 'services' && <ServicesPage navigateTo={navigateTo} />}

        {currentPage.startsWith('method-') && <MethodDetailPage stepId={currentPage.replace('method-', '')} navigateTo={navigateTo} />}

        {currentPage === 'contact' && <ContactPage bookedSlots={bookedSlots} />}

      </main>

      <footer className="py-12 bg-black border-t border-white/5 text-center text-[10px] text-zinc-600 font-bold uppercase tracking-[0.2em] relative z-10">

         © 2026 Nexus Growth System. Tout droits réservés.

      </footer>

    </div>

  );

};
